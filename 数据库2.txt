-- tcm-ai-platform 旧库迁移脚本（老版本 -> 当前用户体系）
-- 适用：已存在 user/consultation_log 老表的情况

USE tcm_ai;

-- 1) 老 user 表重命名为 app_user（若已是 app_user 则跳过）
SET @has_user := (
  SELECT COUNT(*)
  FROM information_schema.tables
  WHERE table_schema = DATABASE() AND table_name = 'user'
);

SET @has_app_user := (
  SELECT COUNT(*)
  FROM information_schema.tables
  WHERE table_schema = DATABASE() AND table_name = 'app_user'
);

SET @sql := IF(@has_user > 0 AND @has_app_user = 0,
  'RENAME TABLE `user` TO `app_user`',
  'SELECT 1'
);
PREPARE stmt FROM @sql;
EXECUTE stmt;
DEALLOCATE PREPARE stmt;

-- 2) 若 app_user 不存在，则补建
CREATE TABLE IF NOT EXISTS `app_user` (
  `id` BIGINT AUTO_INCREMENT PRIMARY KEY,
  `username` VARCHAR(50) NOT NULL UNIQUE COMMENT '用户名',
  `password` VARCHAR(100) NOT NULL COMMENT 'BCrypt 哈希密码',
  `create_time` DATETIME NOT NULL DEFAULT CURRENT_TIMESTAMP COMMENT '注册时间'
) ENGINE=InnoDB DEFAULT CHARSET=utf8mb4 COMMENT='平台用户表';

-- 3) consultation_log 补齐 messages 列
ALTER TABLE `consultation_log`
  ADD COLUMN IF NOT EXISTS `messages` TEXT COMMENT '聊天内容(JSON字符串)' AFTER `title`;

-- 4) 补充索引
ALTER TABLE `consultation_log`
  ADD INDEX IF NOT EXISTS `idx_consultation_user_time` (`user_id`, `create_time`);

-- 5) 外键（如已存在同名外键将自动忽略）
SET @has_fk := (
  SELECT COUNT(*)
  FROM information_schema.table_constraints
  WHERE table_schema = DATABASE()
    AND table_name = 'consultation_log'
    AND constraint_type = 'FOREIGN KEY'
    AND constraint_name = 'fk_consultation_app_user'
);

SET @sql_fk := IF(@has_fk = 0,
  'ALTER TABLE `consultation_log` ADD CONSTRAINT `fk_consultation_app_user` FOREIGN KEY (`user_id`) REFERENCES `app_user` (`id`) ON DELETE CASCADE',
  'SELECT 1'
);
PREPARE stmt_fk FROM @sql_fk;
EXECUTE stmt_fk;
DEALLOCATE PREPARE stmt_fk;

-- 说明：
-- 当前登录使用 BCrypt 哈希密码。老表中若是明文密码，建议让用户重新注册，
-- 或按业务规则离线改写为 BCrypt 哈希。
